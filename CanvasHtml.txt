import org.openqa.selenium.By;  
import org.openqa.selenium.JavascriptExecutor;  
import org.openqa.selenium.WebDriver;  
import org.openqa.selenium.chrome.ChromeDriver;  
  
public class CanvasHtml{  
  
    public static void main(String[] args) {  
  
        System.setProperty("webdriver.chrome.driver", "D:\\ChromeDriver\\chromedriver.exe");  
        WebDriver driver=new ChromeDriver();  
        // Launch Website  
        driver.navigate().to("http://www.htmlcanvasstudio.com/");  
         //Maximize the browser  
        driver.manage().window().maximize();  
         //Find canvas webelement 
        WebElement canvasLine = driver.findElement(By.Xpath("//input[@title="Draw a line"]"));
        canvasLine.click();
          //Find canvas element 
        WebElement canvasElement = driver.findElement(By.id("imageTemp"));

          //Draw the horizontal line
         Actions builder = new Actions(driver);
         builder.moveToElement(canvasElement, 236, 135).click().moveByOffset(238,268).release().perform();
           
    	   //Draw Vertical line
          builder.moveToElement(canvasElement, 151,183).click().moveByOffset(332,-181).release().perform();
 
           //use javascript executor
	  WebElement Rect = driver.findElement(By.Xpath("//input[@title="Draw a rectangle"]"));
	  Rect.click();
	  JavascriptExecutor jsExecutor = (JavascriptExecutor)driver;
          jsExecutor.executorScript("context.strokeRect(86, 347, 278, 102)");

            //Erase horizontal line
          WebElement eraseHorizon= driver.findElement(By.Xpath("//input[@title="Use eraser"]"));
          eraseHorizon.click();	  
	  builder.moveToElement(canvasElement, 236,135).clickAndHold().moveByOffset(238,268).release().perform();

  
    }  
  
}
